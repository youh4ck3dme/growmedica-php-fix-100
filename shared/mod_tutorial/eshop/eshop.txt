<?

	// vlozit do config.inc.php v tomto poradi
	// 146 je menu_id stranky s modulom mod_eshop (tej hlavnej, ktora obsahuje vsetky podsekcie)
	// konstanta ESHOP_MAIN_CATEGORY sa potom pouziva vsade v module, aj v triedach eshopu
	//

	define('ESHOP_MAIN_CATEGORY', 146);
	require_once("classes/class.eshop.php");

	//
	//
	//

	// vlozit PRED HTML
	// obvykle riadok 34 v index.php zo sixadmin
	//
	// kosik je tu zastupeny ako objekt a prenasa sa cez session
	// ale aby sa tak mohlo stat, musis byt najprv do session serializovany ako retazec
	// a ak s nim chceme pracovat, tak ho musime unserializovat a po upravach opat serializovat
	// ale proces serializovania a unserializovania sa sam vykonava pri kazdej operacii, takze s tym netreba velmi pracovat
	// len na to treba mysliet ak budeme robit nejake extra funkcie a operacie 
	//

	if(empty($_SESSION['serialized_cart']))
	{
		$obj_cart = new Cart;
		$_SESSION['serialized_cart'] = serialize($obj_cart);	// toto treba tiez vykonat VZDY po zmene v kosiku (inak sa zmena trvalo neprejavi)
	}
	else
		$obj_cart = unserialize($_SESSION['serialized_cart']);

	// 
	//
	//

	// vlozit do BODY tam kde treba
	//

	echo '<div id="cart_info">';
	echo '<h3>'.$cTranslator->getTranslation('NÁKUPNÝ KOŠÍK:').'</h3><br/>';
	echo '<table>';
	echo '<tr><td>'.$cTranslator->getTranslation('Počet produktov:').'</td><td>'.$obj_cart->get_cart_count().'</td></tr>';
	echo '<tr><td>'.$cTranslator->getTranslation('Celková suma:').'</td><td>'.number_format($obj_cart->get_cart_value(), 2, '.', ' ').' &euro;</td></tr>';
	echo '<tr><td><a href="'.ROOTDIR.'/'.Menu::getHyperlinkById(ESHOP_MAIN_CATEGORY).'/kosik/">'.$cTranslator->getTranslation('Prehľad košíka').'</a></td>
	          <td><a href="'.ROOTDIR.'/'.Menu::getHyperlinkById(ESHOP_MAIN_CATEGORY).'/kosik/step1/">'.$cTranslator->getTranslation('Objednať').'</a></td></tr>';
	echo '</table>';
	echo '</div>';

?>
